    \section{11.1 Sequences}\label{sequences}

\textbf{Sequence}: A sequence can be thought of as a list of numbers
written in a definite order: \[a_1, a_2, a_3, \cdots, a_n, \cdots \]

\begin{itemize}
\tightlist
\item
  \(a_1\): first term,
\item
  \(a_2\): second term,
\item
  \(a_3\): third term,
\item
  \(\vdots\)
\item
  \(a_n\): \(\text{n}^\text{th}\) term,
\end{itemize}

\begin{center}\rule{0.5\linewidth}{\linethickness}\end{center}

\textbf{For example}: * \(1, 2, 3, \cdots\) * \(1, 1/2, 1/3, \cdots\) *
\(-1, 1, -1, \cdots\)

    \subsection{Notation:}\label{notation}

\begin{itemize}
\tightlist
\item
  \(\{a_1, a_2, a_3, \cdots, a_n, \cdots\} = \{a_n\}\) or
\item
  \(\{a_1, a_2, a_3, \cdots, a_n, \cdots \}= \left\{a_n\right\}_{n=1}^{\infty}\)
\end{itemize}

    \subsection{More examples}\label{more-examples}

\begin{itemize}
\tightlist
\item
  \(\left\{\frac{n}{n+1}\right\}\)
\item
  \(\left\{\frac{(-1)^n(n+1)}{5^n}\right\}\)
\item
  \(\left\{\sqrt{n-4}\right\}_{n=4}^{\infty}\)
\item
  \(a_1=1\), \(a_2=1\) , \(a_n=a_{n-1}+a_{n-2}\) (\textbf{Fibonacci
  sequence})
\end{itemize}
